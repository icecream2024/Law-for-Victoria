je_military_school = {
	group = je_group_world_war
	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	
	scripted_progress_bar = military_schools_progress_bar

	complete = {
		OR={
			custom_tooltip = {
				text = je_military_schools_complete_tt
				scope:journal_entry = {
					"scripted_bar_progress(military_schools_progress_bar)" = 120
				}
			}
			AND = {
				OR = {
					has_law = law_type:law_single_party_state
					has_law = law_type:law_autocracy
				}
				OR = {
					has_law = law_type:law_council_republic
					has_law = law_type:law_corporate_state
				}
			}	
		}
	}

	on_complete = {
		trigger_event = {
			id = disc_military_school.2
			popup = yes
		}
	}

	fail = {
		custom_tooltip = {
			text = je_military_schools_fail_tt
			scope:journal_entry = {
				"scripted_bar_progress(military_schools_progress_bar)" = 0
			}
		}
		has_law = law_type:law_peasant_levies
	}
	
	on_fail = {
		trigger_event = {
			id = disc_military_school.3
			popup = yes
		}
	}

	weight = 400
    transferable = no
	should_be_pinned_by_default = yes
}
